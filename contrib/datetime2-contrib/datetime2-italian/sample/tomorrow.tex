\documentclass[italian]{article}
\newcount\myct
\newcount\datecount
\newcommand{\tomorrow}{%
	\DTMsavenow{mydate}
	\DTMsaveddateoffsettojulianday{mydate}{365}{\myct}
	\DTMsavejulianday{mydate}{\number\myct}
	\DTMusedate{mydate}
}

\newcommand{\yesterday}{%
	\DTMsavenow{mydate}
	\DTMsaveddateoffsettojulianday{mydate}{-1}{\myct}
	\DTMsavejulianday{mydate}{\number\myct}
	\DTMusedate{mydate}
}
\newcommand{\myday}[1]{%
	\DTMsavenow{mydate}
	\DTMsaveddateoffsettojulianday{mydate}{#1}{\myct}
	\DTMsavejulianday{mydate}{\number\myct}
	\DTMusedate{mydate}
}
